
1. JWT Token User Table
CREATE TABLE JWT_TOKEN (
  USERNAME VARCHAR(50)	 NOT NULL DEFAULT '',
  TOKEN VARCHAR(255)	 NOT NULL DEFAULT '',
  ISSUED_DATE			 VARCHAR(20) NOT NULL DEFAULT '',
  EXPIRATION_DATE		 VARCHAR(20) NOT NULL DEFAULT '',
  REMOTE_IP				 VARCHAR(255) NOT NULL DEFAULT '',
  STATUS				 VARCHAR(1) NOT NULL DEFAULT '',
  USERTYPE				 VARCHAR(2) DEFAULT '',
  PRIMARY KEY ( USERNAME)
)

2. Sequence Table Master
CREATE TABLE SEQUENCE_NO (
		SEQNO_CODE		VARCHAR(6) 		NOT NULL,
		SEQNO_STATUS	VARCHAR(2) 		NOT NULL DEFAULT ''::CHARACTER VARYING, -- 01: Normal, 00: Deleted
		SEQNO_NAME		VARCHAR(100) 	NOT NULL DEFAULT '':: CHARACTER VARYING, -- Send Telegram SeqNo
		SEQNO_DIGIT		NUMERIC(2) 		NOT NULL,
		SEQNO_START		NUMERIC(16) 	NOT NULL, -- 1
		SEQNO_END		NUMERIC(16) 	NOT NULL, -- 9999
		CREATE_BY		VARCHAR(50)		NOT NULL, -- User that Create
		REGISTER_DATE 	VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
		REGISTER_TIME 	VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
		UPDATE_DATE 	VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
		UPDATE_TIME 	VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
		UPDATE_BY		VARCHAR(50) 	NOT NULL, -- User that update
		CONSTRAINT PK_SEQUENCE_NO PRIMARY KEY ( SEQNO_CODE )
)	


3. Sequence Table Detail
CREATE TABLE SEQUENCE_DETAIL (
	SEQNO_CODE			VARCHAR(6) 		NOT NULL,
	SEQNO_UNIQUE_CODE	VARCHAR(100) 	NOT NULL,
	SEQNO_LAST			NUMERIC(16) 	NOT NULL,
	CREATE_BY		VARCHAR(50)			NOT NULL, -- User that Create
	REGISTER_DATE 	VARCHAR(8) 			NOT NULL DEFAULT ''::CHARACTER VARYING,
	REGISTER_TIME 	VARCHAR(9) 			NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_DATE 	VARCHAR(8) 			NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_TIME 	VARCHAR(9) 			NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_BY		VARCHAR(50) 		NOT NULL, -- User that update
	CONSTRAINT PK_SEQUENCE_DETAIL PRIMARY KEY ( SEQNO_CODE, SEQNO_UNIQUE_CODE )
)


4. User Table
CREATE TABLE USER_INFO(
	USER_NAME 				VARCHAR(20) 	NOT NULL, -- USER NAME FOR LOGIN
	USER_TYPE 				VARCHAR(2) 		NOT NULL DEFAULT ''::CHARACTER VARYING, -- 01: MOBILE, 02: WEB
	USER_STATUS 			VARCHAR(2) 		NOT NULL DEFAULT ''::CHARACTER VARYING, -- 00: LOCK, 01: NORMAL, 99: DELETE
	USER_PASSWORD 			VARCHAR( 255) 	NOT NULL DEFAULT ''::CHARACTER VARYING, 
	PASSWORD_CHANGE_DATE 	VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	PASSWORD_CHANGE_TIME 	VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	FIRST_LOGIN_DATE 		VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	FIRST_LOGIN_TIME 		VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	LAST_LOGIN_DATE 		VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	LAST_LOGIN_TIME 		VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING, 
	MASTER_USER_NAME 		VARCHAR(20) 	NOT NULL DEFAULT ''::CHARACTER VARYING,
	EMPLOYEE_ID				VARCHAR(20) 	NOT NULL DEFAULT ''::CHARACTER VARYING, -- RESERVE COLUMN 
	CREATE_BY				VARCHAR(50)		NOT NULL, -- USER THAT CREATE
	REGISTER_DATE 			VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	REGISTER_TIME 			VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_DATE 			VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_TIME 			VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_BY				VARCHAR(50) 	NOT NULL, -- USER THAT update
	CONSTRAINT PK_USER_INFO PRIMARY KEY (USER_NAME)
)

5. User Table History
CREATE TABLE USER_INFO_HISTORY --- USE FOR RECORD USER CHANGE STATUS , CHANGE PASSWORD 
			(
				USER_NAME 				VARCHAR(20) 	NOT NULL, -- USER NAME FOR LOGIN

				CREATE_BY				VARCHAR(50)		NOT NULL, -- USER THAT CREATE
				REGISTER_DATE 			VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
				REGISTER_TIME 			VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
				UPDATE_DATE 			VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
				UPDATE_TIME 			VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
				UPDATE_BY				VARCHAR(50) 	NOT NULL, -- USER THAT UPDATE

			)


4. Menu Master 

CREATE TABLE MENU_MASTER (
	MENU_SEQNO			NUMERIC(10) 	NOT NULL,
	MENU_CODE   		VARCHAR(20) 	NOT NULL DEFAULT ''::CHARACTER VARYING,
	MENU_DESCRIPTION  	VARCHAR(255) 	NOT NULL DEFAULT ''::CHARACTER VARYING,
	MENU_URL  			VARCHAR(255)	NOT NULL DEFAULT ''::CHARACTER VARYING,
	USE_YN				VARCHAR(1) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	CREATE_BY			VARCHAR(50)		NOT NULL, -- USER THAT CREATE
	REGISTER_DATE 		VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	REGISTER_TIME 		VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_DATE 		VARCHAR(8) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_TIME 		VARCHAR(9) 		NOT NULL DEFAULT ''::CHARACTER VARYING,
	UPDATE_BY			VARCHAR(50) 	NOT NULL, -- USER THAT UPDATE
	CONSTRAINT PK_MENU_MASTER PRIMARY KEY ( MENU_SEQNO )
)
